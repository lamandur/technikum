Урок 5: Работа с файловой системой

Цель урока: Научиться сохранять данные (текст, изображения) на устройство пользователя с помощью файловой системы Cordova.

Требования: Готовый проект Cordova с установленными плагинами камеры и геолокации.

Важно: Весь код работает только в собранном приложении. Тестирование в браузере невозможно.

Зачем нужна файловая система?

localStorage ограничен ~5 МБ и хранит только текст. Файловая система позволяет:

  • Сохранять изображения, аудио, видео
  • Хранить большие объёмы данных
  • Создавать настоящие файлы в памяти устройства

Пошаговая инструкция

Шаг 1: Установка плагина файловой системы

cordova plugin add cordova-plugin-file

Проверьте установку:

cordova plugin list

Шаг 3: JavaScript-код — работа с файлами

Создайте файл www/js/app.js:

// Сохранение текстового файла
function saveTextFile() {
    const filename = document.getElementById('filename').value.trim();
    const content = document.getElementById('content').value.trim();

    if (!filename) {
        alert('Введите имя файла');
        return;
    }

    const fullFilename = filename + '.txt';

    window.resolveLocalFileSystemURL(
        cordova.file.dataDirectory,
        function(dirEntry) {
            dirEntry.getFile(fullFilename, { create: true }, function(fileEntry) {
                fileEntry.createWriter(function(fileWriter) {
                    fileWriter.onwriteend = function() {
                        alert('Файл сохранён: ' + fullFilename);
                    };
                    fileWriter.onerror = function(e) {
                        alert('Ошибка записи: ' + e.toString());
                    };
                    fileWriter.write(new Blob([content], { type: 'text/plain' }));
                });
            });
        },
        function(error) {
            alert('Ошибка доступа к файловой системе');
        }
    );
}

// Загрузка текстового файла
function loadTextFile() {
    const filename = document.getElementById('filename').value.trim();
    if (!filename) {
        alert('Введите имя файла');
        return;
    }

    const fullFilename = filename + '.txt';

    window.resolveLocalFileSystemURL(
        cordova.file.dataDirectory,
        function(dirEntry) {
            dirEntry.getFile(fullFilename, {}, function(fileEntry) {
                fileEntry.file(function(file) {
                    const reader = new FileReader();
                    reader.onloadend = function() {
                        document.getElementById('content').value = this.result;
                    };
                    reader.readAsText(file);
                });
            });
        },
        function(error) {
            alert('Файл не найден: ' + fullFilename);
        }
    );
}

// Сохранение изображения
function saveImageToFile(base64Data, filename) {
    window.resolveLocalFileSystemURL(
        cordova.file.dataDirectory,
        function(dirEntry) {
            dirEntry.getFile(filename, { create: true }, function(fileEntry) {
                fileEntry.createWriter(function(fileWriter) {
                    fileWriter.onwriteend = function() {
                        alert('Изображение сохранено: ' + filename);
                        // Показываем сохранённое фото
                        document.getElementById('savedPhoto').src =
                            "image/jpeg;base64," + base64Data;
                        document.getElementById('savedPhoto').style.display = "block";
                    };
                    fileWriter.onerror = function(e) {
                        alert('Ошибка сохранения изображения');
                    };

                    // Обработка base64
                    const data = base64Data.replace(/^image\/jpeg;base64,/, "");
                    const binary = atob(data);
                    const array = [];
                    for (let i = 0; i < binary.length; i++) {
                        array.push(binary.charCodeAt(i));
                    }
                    const blob = new Blob([new Uint8Array(array)], { type: 'image/jpeg' });
                    fileWriter.write(blob);
                });
            });
        },
        function(error) {
            alert('Ошибка доступа к файловой системе');
        }
    );
}

// Съёмка и сохранение фото
function takeAndSavePhoto() {
    navigator.camera.getPicture(
        function(imageData) {
            const filename = 'photo_' + Date.now() + '.jpg';
            saveImageToFile(imageData, filename);
        },
        function(message) {
            alert('Ошибка камеры: ' + message);
        },
        {
            quality: 60,
            destinationType: Camera.DestinationType.DATA_URL
        }
    );
}

Шаг 4: Сборка и тестирование

# Подключите телефон с отладкой по USB
cordova run android

Или соберите APK:

cordova build android
# Установите APK из: platforms/android/app/build/outputs/apk/debug/app-debug.apk

Структура проекта

my-files/
├── config.xml
├── www/
│   ├── index.html
│   └── js/
│       └── app.js          ← Весь код здесь
├── plugins/
│   └── cordova-plugin-file/
└── platforms/
✅ Важно: Все файлы сохраняются в приватную папку приложения (cordova.file.dataDirectory). Они недоступны пользователю напрямую, но сохраняются между запусками.

Задания

  1. Установите плагин cordova-plugin-file.
  2. Реализуйте сохранение и загрузку текстовых файлов.
  3. Добавьте функцию съёмки и сохранения фото.
  4. Протестируйте всё на реальном устройстве через cordova run android.

Решение типичных проблем

"FileError.NOT_FOUND_ERR" → Убедитесь, что используете { create: true } при создании файла.

Изображение не сохраняется → Проверьте обработку base64-строки (уберите префикс image/jpeg;base64,).

Нет доступа к файлам → Используйте только cordova.file.dataDirectory, другие пути могут не работать.
