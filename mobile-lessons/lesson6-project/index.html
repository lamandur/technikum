// Глобальная переменная для хранения фото
let currentPhoto = null;

// Функция для запуска камеры
function takePhoto() {
    if (!navigator.camera) {
        alert('Камера не доступна. Убедитесь, что работаете в Cordova-приложении!');
        return;
    }

    navigator.camera.getPicture(
        function(imageData) {
            // Успешное получение фото
            currentPhoto = imageData;
            const photoPreview = document.getElementById('photoPreview');
            photoPreview.src = "image/jpeg;base64," + imageData;
            photoPreview.style.display = "block";
        },
        function(message) {
            // Ошибка камеры
            alert('Ошибка камеры: ' + message);
        },
        {
            quality: 60,
            destinationType: Camera.DestinationType.DATA_URL,
            sourceType: Camera.PictureSourceType.CAMERA,
            encodingType: Camera.EncodingType.JPEG
        }
    );
}

// Функция для сохранения заметки
function saveNote() {
    const title = document.getElementById('noteTitle').value.trim();
    const content = document.getElementById('noteContent').value.trim();

    if (!title) {
        alert('Введите заголовок заметки!');
        return;
    }

    // Генерируем уникальное имя файла
    const timestamp = Date.now();
    const noteFilename = `note_${timestamp}.txt`;
    const photoFilename = `photo_${timestamp}.jpg`;

    // Создаем содержимое заметки
    let noteContent = `Заголовок: ${title}\n`;
    noteContent += `Дата: ${new Date().toLocaleString()}\n`;
    if (content) {
        noteContent += `\n${content}\n`;
    }

    // Сохраняем текстовую заметку
    saveTextToFile(noteContent, noteFilename, function() {
        if (currentPhoto) {
            // Сохраняем фото, если оно есть
            saveImageToFile(currentPhoto, photoFilename, function() {
                alert('Заметка с фото сохранена!');
                clearForm();
            });
        } else {
            alert('Текстовая заметка сохранена!');
            clearForm();
        }
    });
}

// Функция для очистки формы
function clearForm() {
    document.getElementById('noteTitle').value = '';
    document.getElementById('noteContent').value = '';
    document.getElementById('photoPreview').style.display = 'none';
    currentPhoto = null;
}

// Функция для сохранения текста в файл
function saveTextToFile(text, filename, callback) {
    window.resolveLocalFileSystemURL(
        cordova.file.dataDirectory,
        function(directory) {
            directory.getFile(filename, { create: true }, function(fileEntry) {
                fileEntry.createWriter(function(fileWriter) {
                    fileWriter.onwriteend = callback;
                    fileWriter.onerror = function(e) {
                        alert('Ошибка записи файла: ' + e.toString());
                    };
                    fileWriter.write(new Blob([text], { type: 'text/plain' }));
                });
            });
        },
        function(error) {
            alert('Ошибка доступа к файловой системе: ' + error.code);
        }
    );
}

// Функция для сохранения изображения в файл
function saveImageToFile(base64Data, filename, callback) {
    window.resolveLocalFileSystemURL(
        cordova.file.dataDirectory,
        function(directory) {
            directory.getFile(filename, { create: true }, function(fileEntry) {
                fileEntry.createWriter(function(fileWriter) {
                    fileWriter.onwriteend = callback;
                    fileWriter.onerror = function(e) {
                        alert('Ошибка сохранения изображения: ' + e.toString());
                    };

                    // Обрабатываем base64 данные
                    const data = base64Data.replace(/^image\/jpeg;base64,/, "");
                    const binary = atob(data);
                    const array = [];
                    for (let i = 0; i < binary.length; i++) {
                        array.push(binary.charCodeAt(i));
                    }
                    const blob = new Blob([new Uint8Array(array)], { type: 'image/jpeg' });
                    fileWriter.write(blob);
                });
            });
        },
        function(error) {
            alert('Ошибка доступа к файловой системе: ' + error.code);
        }
    );
}

// Функция для загрузки всех заметок
function loadAllNotes() {
    const notesList = document.getElementById('notesList');
    notesList.innerHTML = '<p>Загрузка...</p>';

    window.resolveLocalFileSystemURL(
        cordova.file.dataDirectory,
        function(directory) {
            const reader = directory.createReader();
            reader.readEntries(function(entries) {
                // Фильтруем только текстовые файлы заметок
                const noteFiles = entries.filter(entry =>
                    entry.name.startsWith('note_') && entry.name.endsWith('.txt')
                );

                if (noteFiles.length === 0) {
                    notesList.innerHTML = '<p>Нет сохранённых заметок</p>';
                    return;
                }

                // Сортируем по дате (новые сверху)
                noteFiles.sort((a, b) => {
                    const timeA = parseInt(a.name.replace('note_', '').replace('.txt', ''));
                    const timeB = parseInt(b.name.replace('note_', '').replace('.txt', ''));
                    return timeB - timeA;
                });

                // Читаем все заметки
                let notesHtml = '';
                let processed = 0;

                noteFiles.forEach(fileEntry => {
                    fileEntry.file(function(file) {
                        const reader = new FileReader();
                        reader.onloadend = function() {
                            const content = this.result;
                            const lines = content.split('\n');
                            const title = lines[0].replace('Заголовок: ', '');
                            const date = lines[1].replace('Дата: ', '');

                            // Ищем соответствующее фото
                            const timestamp = fileEntry.name.replace('note_', '').replace('.txt', '');
                            const photoPath = cordova.file.dataDirectory + 'photo_' + timestamp + '.jpg';

                            notesHtml += `
                                <div class="note-item">
                                    <div class="note-title">${title}</div>
                                    <div class="note-meta">${date}</div>
                                    <div class="note-content">${lines.slice(3).join('\n')}</div>
                            `;

                            // Проверяем существование фото (упрощённо)
                            notesHtml += `</div>`;

                            processed++;
                            if (processed === noteFiles.length) {
                                notesList.innerHTML = notesHtml;
                            }
                        };
                        reader.readAsText(file);
                    });
                });
            });
        },
        function(error) {
            notesList.innerHTML = '<p>Ошибка загрузки заметок</p>';
        }
    );
}

Частые проблемы и решения

Проблема: Приложение не запускается на устройстве

Решение: Убедитесь, что включена "Отладка по USB" в настройках разработчика.

Проблема: Камера не работает

Решение: Проверьте разрешения в настройках приложения на телефоне.

Проблема: Файлы не сохраняются

Решение: Убедитесь, что используете cordova.file.dataDirectory и обрабатываете ошибки.
